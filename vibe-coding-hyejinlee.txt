      if (prevIdx < 0) {

      playSubTrack(prevIdx);

    function togglePlay() {
      if (isPlaying) {
        isPlaying = false;
        playPauseIcon.setAttribute('data-lucide', 'play');
        widgetVinyl.style.animationPlayState = 'paused';
      } else {
        isPlaying = true;

        widgetVinyl.style.animationPlayState = 'running';
      }
      lucide.createIcons();
    }

    // 가상 탐색 바 제어
    function updateProgressBar() {
      clearInterval(progressTimer);
      progressTimer = setInterval(() => {
        if (!isPlaying) return;
        currentSeconds++;
        const totalStr = playlistsData[currentPlaylistIndex].tracks[currentTrackIndex].time;
        if (totalStr === "00:00") {
          progressFill.style.width = `0%`;
          playerCurrentTime.innerText = "0:00";
          return;
        }
        
        const totalParts = totalStr.split(':');
        const totalSecs = parseInt(totalParts[0]) * 60 + parseInt(totalParts[1]);

        if (currentSeconds >= totalSecs) {
          nextTrack(); // 재생이 끝나면 다음곡으로 자동 이동 시뮬레이션
          return;
        }

        const percentage = (currentSeconds / totalSecs) * 100;
        progressFill.style.width = `${percentage}%`;

        const m = Math.floor(currentSeconds / 60);
        const s = currentSeconds % 60;
        playerCurrentTime.innerText = `${m}:${s < 10 ? '0' + s : s}`;
      }, 1000);
    }

    function escapeHtml(text) {
      const map = {
        '&': '&amp;',
        '<': '&lt;',
        '>': '&gt;',
        '"': '&quot;',
        "'": '&#039;'
      };
      return text.replace(/[&<>"']/g, function(m) { return map[m]; });
    }

        name: nameVal,
        emoji: emojiVal.substring(0, 2),
        content: contentVal,
        date: formattedDate
      };

      const comments = getComments();
      comments.unshift(newComment);
      saveComments(comments);
      renderComments();

      commentorName.value = '';
      commentContent.value = '';
    });
